1function [Z,F,B] = tauchen_MY(cover,sige, rho_e, mu, numz, lbb, ubb, numb)
 2
 3% Tauchen discretization of TFP for Mendoza & Yue
 4%Preallocating matrices to speed up the code
 5Z = zeros(numz,1); %discrete spanning space of TFP (I'm calling TFP in level z, log TFP e)
 6F = zeros(numz,numz); %Markov chain transition probability matrix
 7
 8lbz = mu - cover*sige/sqrt((1-rho_e^2)); %lower bound of the z grid
 9ubz = mu + cover*sige/sqrt((1-rho_e^2)); %upper blund of the z grid
10dist = ubz - lbz; %overall distance to be covered by the z grid
11stepsize = dist/(numz-1); %size of each bin of the z grid
12for i = 1:numz
13    Z(i) = lbz + (i-1)*stepsize; %linearly spaced points
14end
15for i = 1:numz-1
16    m(i) = (Z(i) + Z(i+1))/2; %midpoints for the discrete normal CDF
17end
18for j = 1:numz
19    a(j) = rho_e*Z(j); %this is the conditional mean to be in the previous period
20    for i = 1:numz
21        if i == 1
22            F(i,j) = cdf('norm',m(i),a(j),sige); %first point - lowest z'
23        elseif i == numz;
24            F(i,j) = 1 - cdf('norm',m(i-1),a(j),sige); %last point - highest z'
25        else
26            F(i,j) = cdf('norm',m(i),a(j),sige) - cdf('norm',m(i-1),a(j),sige); %middle points
27        end
28    end
29end
30
31% Discretization for bonds (b)
32B=linspace(lbb,ubb,numb)';
